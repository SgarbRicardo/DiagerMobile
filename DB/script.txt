SELECT P.PROD_DESCRICAO, P.PROD_VERSAO, P.PROD_MARCA,P.PROD_FOTO, P.PROD_SNAP_TAG, C.CAT_DESCRICAO, E.EMPRE_lOCATARIO,
       E.EMPRE_IND_SINC, E.EMPRE_ID, E.EMPRE_DT_LOCACAO, E.EMPRE_QTD, U.USU_NOME, U.USU_DEPARTAMENTO
             FROM TAB_EMPRESTIMO E
               JOIN TAB_PRODUTO P ON(E.EMPRE_PROD_ID = P.PROD_ID)
               JOIN TAB_USUARIO U ON (E.EMPRE_LOGIN_ID = U.COD_USUARIO)
               JOIN TAB_CATEGORIA C ON (P.PROD_ID_CATEGORIA = C.ID_CATEGORIA)
               WHERE 1 = 1
               AND E.EMPRE_ID = :arg0
               AND E.EMPRE_PROD_ID = :arg1
